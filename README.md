For the corn and soybean data
==============================

	        j = month, {1,2...,6}, thus April - Sept

e_4_j = tmin

rs_i_j = where i = remote sensing variable {1,2..,5}
	        j = month, {1,2...,6}, thus April - Sept
 
rs_1_j = LST Day
rs_2_j = LST Night
rs_3_j = SR b01
rs_4_j = SR b02
rs_5_j = SR b07
